const API_BASE_URL = "https://svgogh.onrender.com/api"; document.addEventListener('DOMContentLoaded', () => { const signupForm = document.getElementById('signup-form'); const messageArea = document.getElementById('message-area'); //exit if the form isn't found if (!signupForm) return; signupForm.addEventListener('submit', async (e) => { e.preventDefault(); messageArea.textContent = ''; messageArea.classList.remove('text-red-500', 'text-green-500'); // Clear any previous styles const username = signupForm.username.value; const password = signupForm.password.value; // get the value of the new 'password-confirm' input using its name attribute const passwordConfirm = signupForm['password-confirm'].value; if (!username || !password || !passwordConfirm) { messageArea.textContent = 'Please fill in all fields.'; messageArea.classList.add('text-red-500'); return; } if (password !== passwordConfirm) { messageArea.textContent = 'Error: The password and confirmation password do not match.'; messageArea.classList.add('text-red-500'); return; // Stop the function here, do not send to the API } // If validation passes, proceed with API call try { messageArea.textContent = 'Registering...'; const response = await fetch(`${API_BASE_URL}/register`, { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ username, password }), }); const data = await response.json(); if (response.ok) { //success signal messageArea.textContent = 'Registration successful! Redirecting to login...'; messageArea.classList.add('text-green-500'); // Redirect user to the login page after successful registration setTimeout(() => { window.location.href = '/login.html'; }, 1000); } else { //fail signal messageArea.textContent = `Registration failed: ${data.msg || 'Check server status.'}`; messageArea.classList.add('text-red-500'); } } catch (error) { console.error('Registration Network Error:', error); messageArea.textContent = 'Failed to connect to the server.'; messageArea.classList.add('text-red-500'); } }); });
